```text
    id : template_1
    category_id : 1
    description : Template เกี่ยวกับอะไร
    version : 1.0
    header : 1
    footer : 1
    menu : 1-main
    topmenu : 1
    use_sidebar : 1
    theme : theme-style-1
    use_form : true
    banner : 1
    pages : Home, About

 ```
